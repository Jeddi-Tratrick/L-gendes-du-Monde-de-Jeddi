**Arborescence du projet :** ``` / (racine du projet) |-- frontend/ | |-- index.html | |-- style.css |-- server.js |-- package.json ``` **frontend/index.html** ```html Légendes du Monde ``` **server.js** (backend Node.js Express pour Render) ```javascript import express from "express"; import path from "path"; import { fileURLToPath } from "url"; const __filename = fileURLToPath(import.meta.url); const __dirname = path.dirname(__filename); const app = express(); const PORT = process.env.PORT || 3000; app.use(express.json()); app.use(express.static(path.join(__dirname, "frontend"))); // Endpoint pour recevoir les légendes app.post("/api/legendes", (req, res) => { console.log("Nouvelle légende reçue :", req.body); res.status(200).json({ message: "Légende reçue avec succès" }); }); app.listen(PORT, () => { console.log(`Serveur démarré sur le port ${PORT}`); }); ``` **package.json** ```json { "name": "legendes-du-monde", "version": "1.0.0", "main": "server.js", "type": "module", "scripts": { "start": "node server.js" }, "dependencies": { "express": "^4.18.2" } } ```
